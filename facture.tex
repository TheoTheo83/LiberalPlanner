\documentclass{letter}
\usepackage{invoice}
\address{**data['My Name']**  \\ **data['My Street']** \\ **data['My Zip']** \\ **data['my@mail.com']**}
\signature{**data['Fnord Prefake']**}
\date{**data['23. May 2009']**}

\begin{document}
  \begin{letter}{**data['Customer name']** \\ **data['Customer street']** \\ **data['Customer ZIP']**}
    \opening{**data['Invoice no. 1']**}
    **data['My greeting']**
    \begin{invoice}{Euro}{0}
      \ProjectTitle{**data['Example Project']**}%
      \Fee{**data['Development']**} {**data['1000.00']**} {**data['1']**}
    \end{invoice}
    \closing{**data['My Closing']**}
  \end{letter}
\end{document}
